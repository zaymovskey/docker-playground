```bash
(Создать сеть)
docker network create docker-course-network

(Собрать образы)
docker build -t docker-course-server ./server
docker build -t docker-course-client ./client

(Запустить сервер в фоне)
docker run -d --name server --network docker-course-network docker-course-server

(Запустить клиент. После отработки скрипта контейнер остановится)
docker run --network docker-course-network docker-course-client
```
